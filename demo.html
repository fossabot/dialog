Third example

Advanced use: using callbacks, ids and vars.

<script type="text/javascript">
function check(box1) {
	if (box1.action) {
		box1.vars.name = box1.value != "" ? box1.value : "Unnamed";
		dialog({
			type: "confirm",
			content: "Do you want your name printed on a message?",
			vars: box1.vars,
		})
		.then(function(box2) {
			if (box2.action) {
				dialog({
					//NB: default type is alert, so we don't need to specify it
					title: "Print out",
					content: "Hello "+box2.vars.name+"\nFoo: "+box2.vars.foo
				});
			}
		});
	}
}
</script>
